switch

语法格式

switch(expression){
    case expression#1:
        statement
        break
    case expression#2:
        statement
        break
    case expression#3:
        statement
        break
    default expression#4:
        statement
        break
}
switch将圆括号中的表达式与case后面的表达式进行比较
如case后面是字符串或数字，则比较方式是精确匹配
如case后面是正则表达式，则比较方式是正则
如case后面是list、map、range，则比较方式是成员方式
  • 使用正则的例子
    import java.util.regex.Matcher
map2 = [1: 'one', 2: 'two', 3: 'three', 4: 'four', 5: 'five',
        6: 'six', 7: 'seven', 8: 'eight', 9: 'nine', 10: 'ten',
        11: 'eleven', 12: 'twelve', 0: 'midnight']
def hour = 0
switch('00:12:30'){ 
    case ~/([0-9]{2}):10:00/:
        hour = Matcher.lastMatcher[0][1].toInteger()
        //println hour
        hour = hour>12?hour-12:hour
        println "ten past ${map2[hour]}"
        break

    case ~/([0-9]{2}):1[1-4]:[012][0-9]/:
        hour = Matcher.lastMatcher[0][1].toInteger()
        hour = hour>12?hour-12:hour
        println "just after ten past ${map2[hour]}"
        break

    case ~/([0-9]{2}):1[1-4]:[345][0-9]/:
        hour = Matcher.lastMatcher[0][1].toInteger()
        hour = hour>12?hour-12:hour
        println "just before quarter past ${map2[hour]}"
        break

    case ~/([0-9]{2}):15:00/:
        hour = Matcher.lastMatcher[0][1].toInteger()
        hour = hour>12?hour-12:hour
        println "quarter past ${map2[hour]}"
        break

    case ~/([0-9]{2}):30:00/:
        hour = Matcher.lastMatcher[0][1].toInteger()
        hour = hour>12?hour-12:hour
        println "half past ${map2[hour]}"
        break
    
    case ~/([0-9]{2}):45:00/:
        hour = Matcher.lastMatcher[0][1].toInteger()
        hour = hour>12?hour-12:hour
        hour += 1
        println "quarter to ${map2[hour]}"
        break

    case ~/([0-9]{2}):50:00/:
        hour = Matcher.lastMatcher[0][1].toInteger()
        hour = hour>12?hour-12:hour
        hour += 1
        println "ten to ${map2[hour]}"
        break
}

def getBranch(String branchName){
    def matcher = (branchName =~ "RELEASE-[0-9]{4}")

        newBranchName = branchName
    }
