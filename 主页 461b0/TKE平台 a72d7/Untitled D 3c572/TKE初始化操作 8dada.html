硬件需求

#检查CPU数量8核,检查内存大于16G#检查内核,centos7需大于3.10.0-1062
echo -e "CPU内核总数为:\t $(egrep 'processor' /proc/cpuinfo | wc -l)"
echo -e "内存总数为:\t $(free -mk | awk '/^Mem/{print $2}')"
echo -e "内核版本为:\t $(uname -r)"
#划分磁盘，格式化
pvcreate /dev/sdb
vgcreate vg1 /dev/sdb
lvcreate -L 100G vg1 -n cpass
lvcreate -L 49.9G vg1 -n docker

mkfs.xfs /dev/mapper/vg1-cpass && mkfs.xfs /dev/mapper/vg1-docker
mkdir /var/lib/docker && mkdir /cpaas

echo '/dev/mapper/vg1-cpass  /cpaas  xfs  defaults  0 0' >>/etc/fstab
echo '/dev/mapper/vg1-docker  /var/lib/docker  xfs  defaults  0 0' >>/etc/fstab

mount -a

软件需求

#解决 kmem&&关闭大页，找到GRUB_CMDLINE_LINUX= ，然后在 crashkernel=auto后面添加cgroup.memory=nokmem transparent_hugepage=never。
sed -ri '/^GRUB_CMDLINE_LINUX=/s#(crashkernel=auto)#\1 cgroup.memory=nokmem transparent_hugepage=never#' /etc/default/grub
egrep 'crashkernel=auto cgroup.memory=nokmem transparent_hugepage=never ' /etc/default/grub && echo "OK"

#关闭swap
swapoff -a
sed -i -r '/swap/s/^/#/' /etc/fstab
(( $(free -m | awk '/^Swap/{print $2}') == 0 )) && echo "OK"

#关闭防火墙
systemctl disable --now firewalld
systemctl is-enabled firewalld &>/dev/null || echo 'OK'
#关闭selinux
sed -ri 's#^SELINUX=enforcing#SELINUX=disabled#' /etc/selinux/config
setenforce 0 && egrep '^SELINUX' /etc/selinux/config &>/dev/null && echo 'OK'
#时间同步
yum install -y chrony
systemctl enable --now chronyd
sed -i -e '/^server/s/^/#/' -e '1a server ntp.aliyun.com iburst' /etc/chrony.conf
systemctl restart chronyd && echo 'OK'
#设置时区
timedatectl set-timezone Asia/Shanghai 

#修改内核参数,vm.max_map_count是 es 运行的服务器的要求,net.ipv4.ip_forward是 Kubernetes 要求,关闭 file cache  https://cloud.tencent.com/developer/article/1637682
echo 'vm.max_map_count=262144' >> /etc/sysctl.conf
echo 'net.ipv4.ip_forward = 1' >> /etc/sysctl.conf
echo 'vm.drop_caches=3' >> /etc/sysctl.conf
sysctl -p && echo '内核修改完成'

#某些情况下，pod内的进程重启，会在pod内创建core文件，大量占用磁盘空间，最终 pod 挂掉，甚至拖累宿主机,-c是指内核文件。
ulimit -c 0 && echo 'ulimit -S -c 0' >>/etc/profile && echo "已修改ulimit"

#添加内核模块
modprobe iptable_nat && echo iptable_nat >> /etc/modules-load.d/cpaas.conf && lsmod | grep '^iptable_nat'

#安装依赖的程序
yum install -y jq sshpass socat bind-utils net-tools ntpdate

rm -rf /etc/yum.repos.d/*.repo
curl -L http://mirrors.aliyun.com/repo/Centos-7.repo -o /etc/yum.repos.d/Centos-7.repo
curl -L http://mirrors.aliyun.com/repo/epel-7.repo -o /etc/yum.repos.d/epel-7.repo
yum makecache
