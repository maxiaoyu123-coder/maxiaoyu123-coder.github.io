创建NFS的PV和PVC

#本示例选择了K8S集群内的一个节点作为NFS服务器。
yum install -y nfs-utils
mkdir /cpaas/data/nfs
cat <<EOF >/etc/exports
/cpaas/data/nfs	10.200.40.0/24(rw,sync,no_root_squash)
EOF
systemctl enable --now nfs-server
showmount -e
mkdir /root/volume
cat <<EOF >>/etc/fstab
10.200.40.94:/cpaas/data/nfs /root/volume nfs default 0 0
EOF
mount -a
